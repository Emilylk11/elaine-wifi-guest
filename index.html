/* ========================= Guestbook ========================= */

const GB_KEY = 'emily_guestbook_v1';

const defaultSeeds = [

  {name:'Mama', message:'Best Wi-Fi in Texas!'},

  {name:'Sam', message:'If you can read this, bring snacks 😄'},

  {name:'Nikki Wikki', message:'Scary movie night queue: Jeepers Creepers, where\'d ya get those peepersss.'},

  {name:'Imaginary Roomie', message:'Password strength: immaculate.'},

  {name:'Dad', message:'Leave the A/C alone.'},

];

function getNotes(){ try{ const raw = localStorage.getItem(GB_KEY); return raw ? JSON.parse(raw) : defaultSeeds; }catch(e){ return defaultSeeds; } }

function setNotes(arr){ localStorage.setItem(GB_KEY, JSON.stringify(arr.slice(0,100))); }

function addNote(name, message){ const notes = getNotes(); notes.unshift({name:name.trim(), message:message.trim()}); setNotes(notes); renderWall(); }

function renderWall(){

  const notes = getNotes();

  $('#gb-chip-count').innerHTML = `<span class="chip">${notes.length} note${notes.length!==1?'s':''}</span>`;

  const rotator = $('#gb-rotator'); rotator.innerHTML = '';

  notes.slice(0,6).forEach((n,i)=>{

    const div = document.createElement('div');

    div.className = 'wall__msg' + (i===0?' active':'');

    div.innerHTML = `<div class="msg"><div>${escapeHtml(n.message)}</div><small>— ${escapeHtml(n.name)}</small></div>`;

    rotator.appendChild(div);

  });

}

function rotateWall(){

  const slides = $$('.wall__msg', $('#gb-rotator')); if (slides.length<=1) return;

  let idx = slides.findIndex(el => el.classList.contains('active'));

  slides[idx].classList.remove('active'); slides[(idx+1)%slides.length].classList.add('active');

}

let wallTimer = setInterval(rotateWall, 3500);

function escapeHtml(str){ return str.replace(/[&<>"']/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;', "'":'&#39;'}[m])); }

const QUIZ = [

  { q:'Who is Emily\'s favorite artist?', a:['Tate McRae','Ariana Grande','Justin Bieber','The Kid Laroi'], c:2 },

  { q:'What is my favorite food?', a:['Soul Food','Pizza','Bread','Pasta'], c:0 },

let qi=0, score=0, selected=null;

function renderQuestion(){

  const stage = $('#quiz-stage'); const item = QUIZ[qi]; const total = QUIZ.length;

  $('#quiz-bar').style.width = `${(qi/total)*100}%`; $('#quiz-next').disabled = true; $('#quiz-feedback').textContent = '';

  stage.innerHTML = `

    <div class="q">

      <div class="row"><b>Question ${qi+1} of ${total}</b></div>

      <div class="answers" role="group" aria-label="Answers">

  $$('.answer input', stage).forEach(input=>{ input.addEventListener('change', e=>{ selected=parseInt(e.target.value,10); $('#quiz-next').disabled=false; }); });

}

renderQuestion();

$('#quiz-next').addEventListener('click', ()=>{

  if (selected===null) return;

  const correct = QUIZ[qi].c; if (selected===correct){ score++; showFeedback(true);} else { showFeedback(false); }

$('#quiz-restart').addEventListener('click', ()=>{ qi=0; score=0; selected=null; renderQuestion(); });

function showFeedback(ok){ const fb=$('#quiz-feedback'); fb.className='notice '+(ok?'ok':'nope'); fb.textContent= ok ? 'Correct! 🔥' : 'Close! Keep going 💪'; }

  const stage = $('#quiz-stage'); $('#quiz-bar').style.width = `100%`; const pass = score>=3;

  stage.innerHTML = `

  if (pass) launchConfetti(850);

  $('#quiz-restart-cta')?.addEventListener('click', ()=>{ qi=0; score=0; selected=null; renderQuestion(); });

}


/* ========================= Confetti ========================= */

const confettiCanvas = $('#confetti'), ctx = confettiCanvas.getContext('2d');

let confettiPieces=[], confettiRunning=false;

function resizeCanvas(){ confettiCanvas.width=window.innerWidth; confettiCanvas.height=window.innerHeight; }

window.addEventListener('resize', resizeCanvas); resizeCanvas();

function launchConfetti(n=600){

  for (let i=0;i<n;i++){

    confettiPieces.push({ x:Math.random()*confettiCanvas.width, y:-10-Math.random()*40, r:2+Math.random()*4, a:Math.random()*Math.PI*2, s:1+Math.random()*2.5, w:.5+Math.random()*1.5, c: randomColor() });

  }

  if (!confettiRunning){ confettiRunning=true; requestAnimationFrame(tickConfetti); setTimeout(()=>{ confettiPieces=[]; confettiRunning=false; clearCanvas(); }, 5000); }

}

function randomColor(){ const palette=['#FF7DAF','#FF9FB2','#F8D9E6','#FFDDE8','#67DEC7']; return palette[(Math.random()*palette.length)|0]; }

function clearCanvas(){ ctx.clearRect(0,0,confettiCanvas.width, confettiCanvas.height); }

function tickConfetti(){

  clearCanvas();

  confettiPieces.forEach(p=>{ p.y+=p.s; p.x+=Math.sin(p.a)*p.w; p.a+=0.05;

    ctx.beginPath(); ctx.fillStyle=p.c; ctx.arc(p.x,p.y,p.r,0,Math.PI*2); ctx.fill();

  });

  confettiPieces = confettiPieces.filter(p=> p.y < confettiCanvas.height + 10);

  if (confettiRunning) requestAnimationFrame(tickConfetti);

}

    if (!el) return; e.preventDefault(); el.scrollIntoView({behavior:'smooth'});

  });

});
